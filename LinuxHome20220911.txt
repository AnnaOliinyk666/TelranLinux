#!/bim/bash
for run in {1..10}
touch $run.txt
sleep 2
done

#!/bim/bash - ШЕБАНГ показывает, чем открывать наш файл
for run in {1..10} - начало цикла. мы говорим, что мы заводим новую переменную, 
которая называется run и она принимает значение от 1 до 10.
touch $run.txt - команда которая будет выполнятся в цикле. Мы говорим что хотим 
создать файл название которого будет принимать значение нашей переменной.
sleep 2 - мы говорим системе сколько нужно подождать прежде чем 
приступить к следующей итерации цикла в секундах.
done - мы говорим программе что это последняя команда в нашем цикле.
